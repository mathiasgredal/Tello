178  // str_ext must map dectype(*c_beg) to std::string_view
179  // transform is whatever transformation you want to do to the matching elements.
180  template <typename ForwardIt, typename StrExtractor = identity, typename Transform = identity>
181  auto prefix_search(std::string_view prefix, ForwardIt c_beg, ForwardIt c_end,
182  StrExtractor&& str_ext = {}, Transform&& transform = {}) -> std::vector<std::decay_t<decltype(transform(*c_beg))>> {
183 
184  auto lower = std::lower_bound(c_beg, c_end, prefix);
185  auto higher = std::upper_bound(lower, c_end, prefix, [&str_ext](std::string_view pre, const auto& element) {
186  std::string_view val = str_ext(element);
187  return pre.substr(0, std::min(val.size(), pre.size())) < val.substr(0, std::min(val.size(), pre.size()));
188  });
189 
190  std::vector<std::decay_t<decltype(transform(*c_beg))>> ans;
191  std::transform(lower, higher, std::back_inserter(ans), std::forward<Transform>(transform));
192  return ans;
193  }
