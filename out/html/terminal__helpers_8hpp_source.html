113  static void echo(argument_type& arg) {
114  std::string str{};
115  str = arg.command_line[1];
116  for (auto it = std::next(arg.command_line.begin(), 2) ; it != arg.command_line.end() ; ++it) {
117  str.reserve(str.size() + it->size() + 1);
118  str += ' ';
119  str += *it;
120  }
121  message msg;
122  msg.value = std::move(str);
123  msg.color_beg = msg.color_end = 0; // color is disabled when color_beg == color_end
124  // other parameters are ignored
125  arg.term.add_message(std::move(msg));
126  }

284  constexpr message::severity::severity_t to_imterm_severity(spdlog::level::level_enum lvl) {
285  assert(lvl != spdlog::level::off);
286  if constexpr (
287  ImTerm::message::severity::trace == static_cast<int>(spdlog::level::trace) &&
288  ImTerm::message::severity::debug == static_cast<int>(spdlog::level::debug) &&
289  ImTerm::message::severity::info == static_cast<int>(spdlog::level::info) &&
290  ImTerm::message::severity::warn == static_cast<int>(spdlog::level::warn) &&
291  ImTerm::message::severity::err == static_cast<int>(spdlog::level::err) &&
292  ImTerm::message::severity::critical == static_cast<int>(spdlog::level::critical)) {
293  return static_cast<message::severity::severity_t>(lvl);
294  } else {
295  switch (lvl) {
296  case spdlog::level::trace: return ImTerm::message::severity::trace;
297  case spdlog::level::debug: return ImTerm::message::severity::debug;
298  case spdlog::level::info: return ImTerm::message::severity::info;
299  case spdlog::level::warn: return ImTerm::message::severity::warn;
300  case spdlog::level::err: return ImTerm::message::severity::err;
301  case spdlog::level::critical: return ImTerm::message::severity::critical;
302  default:
303  assert(false);
304  return {};
305  }
306  }
307  }
